def naive_match(p, t): 
    assert len(p) <= len(t)  # –ø—Ä–∏–ø—É—Å–∫–∞—î–º–æ, —â–æ –¥–æ–≤–∂–∏–Ω–∞ —à–∞–±–ª–æ–Ω—É –Ω–µ –±—ñ–ª—å—à–∞ –∑–∞ –¥–æ–≤–∂–∏–Ω—É —Ç–µ–∫—Å—Ç—É 
    occurrences = [] 
    for i in range(0, len(t)-len(p)+1): # –¥–ª—è –∫–æ–∂–Ω–æ–≥–æ –≤–∏—Ä—ñ–≤–Ω—é–≤–∞–Ω–Ω—è —à–∞–±–ª–æ–Ω—É –Ω–∞ —Ç–µ–∫—Å—Ç 
        match = True  # –ø—Ä–∏–ø—É—Å–∫–∞—î–º–æ, —â–æ —î –∑–±—ñ–≥, –¥–æ–∫–∏ –Ω–µ –¥–æ–≤–µ–¥–µ–Ω–æ –∑–≤–æ—Ä–æ—Ç–Ω—î 
        for j in range(0, len(p)):  # –¥–ª—è –∫–æ–∂–Ω–æ—ó –ø–æ–∑–∏—Ü—ñ—ó —à–∞–±–ª–æ–Ω—É 
            if t[i+j] != p[j]: 
                match = False  # –ø—Ä–∏–Ω–∞–π–º–Ω—ñ –æ–¥–∏–Ω —Å–∏–º–≤–æ–ª –Ω–µ –∑–±—ñ–≥–∞—î—Ç—å—Å—è 
                break 
        if match: 
            occurrences.append(i) 
    return occurrences

def z_func(s): 
    # –Ü–Ω—ñ—Ü—ñ–∞–ª—ñ–∑—É—î–º–æ –º–∞—Å–∏–≤ Z, –≤ —è–∫–æ–º—É –±—É–¥–µ–º–æ –∑–±–µ—Ä—ñ–≥–∞—Ç–∏ –∑–Ω–∞—á–µ–Ω–Ω—è Z-—Ñ—É–Ω–∫—Ü—ñ—ó 
    Z = [len(s)] + [0] * (len(s) - 1) # –í—Å—Ç–∞–Ω–æ–≤–ª—é—î–º–æ –¥–æ–≤–∂–∏–Ω—É —Ä—è–¥–∫–∞ s —É –ø–µ—Ä—à–æ–º—É –µ–ª–µ–º–µ–Ω—Ç—ñ –º–∞—Å–∏–≤—É Z, –∞ —Ä–µ—à—Ç—É –∑–∞–ø–æ–≤–Ω—é—î–º–æ –Ω—É–ª—è–º–∏ 
    
    assert len(s) > 1  # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —â–æ –¥–æ–≤–∂–∏–Ω–∞ —Ä—è–¥–∫–∞ s –±—ñ–ª—å—à–µ 1

    # –ü–æ—á–∞—Ç–∫–æ–≤–µ –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è s[1:] –∑ –ø—Ä–µ—Ñ—ñ–∫—Å–æ–º s[0:]
    for i in range(1, len(s)):
        # –ü–æ—Ä—ñ–≤–Ω—é—î–º–æ —Å–∏–º–≤–æ–ª–∏ s[i] —Ç–∞ s[i-1]. –Ø–∫—â–æ –æ–¥–Ω–∞–∫–æ–≤—ñ ‚Äî –∑–±—ñ–ª—å—à—É—î–º–æ Z[1]
        if s[i] == s[i - 1]:
            Z[1] += 1
        else:
            break  # –ü—Ä–∏ –ø–µ—Ä—à—ñ–π –Ω–µ–≤—ñ–¥–ø–æ–≤—ñ–¥–Ω–æ—Å—Ç—ñ ‚Äî –∑—É–ø–∏–Ω—è—î–º–æ—Å—å

    # –í—Å—Ç–∞–Ω–æ–≤–ª—é—î–º–æ –º–µ–∂—ñ –ø—Ä–∞–≤–æ—ó —á–∞—Å—Ç–∏–Ω–∏ (r) —ñ –ª—ñ–≤–æ—ó (l) –∑–±—ñ–≥—É
    r, l = 0, 0
    if Z[1] > 0:
        l, r = 1, Z[1]

    # –û–±—á–∏—Å–ª–µ–Ω–Ω—è Z-—Ñ—É–Ω–∫—Ü—ñ—ó –¥–ª—è –∫–æ–∂–Ω–æ–≥–æ k –≤—ñ–¥ 2 –¥–æ len(s) - 1
    for k in range(2, len(s)):
        assert Z[k] == 0  # –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞, —â–æ –∑–Ω–∞—á–µ–Ω–Ω—è Z[k] —Å–ø–æ—á–∞—Ç–∫—É –¥–æ—Ä—ñ–≤–Ω—é—î 0 

        if k > r:
            # –í–∏–ø–∞–¥–æ–∫ 1: k –∑–∞ –º–µ–∂–∞–º–∏ –ø–æ—Ç–æ—á–Ω–æ–≥–æ –¥—ñ–∞–ø–∞–∑–æ–Ω—É [l, r] 
            for i in range(k, len(s)):
                # –ü–æ—Ä—ñ–≤–Ω—é—î–º–æ —Å–∏–º–≤–æ–ª–∏ s[i] —Ç–∞ s[i-k]. –Ø–∫—â–æ –≤–æ–Ω–∏ —Å–ø—ñ–≤–ø–∞–¥–∞—é—Ç—å, –∑–±—ñ–ª—å—à—É—î–º–æ –∑–Ω–∞—á–µ–Ω–Ω—è Z[k] 
                if s[i] == s[i - k]:
                    Z[k] += 1
                else:
                    break
            # –û–Ω–æ–≤–ª—é—î–º–æ –∑–Ω–∞—á–µ–Ω–Ω—è –∑–º—ñ–Ω–Ω–∏—Ö r —Ç–∞ l 
            l, r = k, k + Z[k] - 1
        else:
            # –í–∏–ø–∞–¥–æ–∫ 2: k –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ –ø–æ—Ç–æ—á–Ω–æ–≥–æ [l, r]
            # –û–±—á–∏—Å–ª—é—î–º–æ –¥–æ–≤–∂–∏–Ω—É –±–µ—Ç–∞ (nbeta) —Ç–∞ –∑–Ω–∞—á–µ–Ω–Ω—è Z-—Ñ—É–Ω–∫—Ü—ñ—ó –Ω–∞ –ø–æ–ø–µ—Ä–µ–¥–Ω—å–æ–º—É –∫—Ä–æ—Ü—ñ (Zkp)
            nbeta = r - k + 1
            Zkp = Z[k - l]
            if nbeta > Zkp:
                # –ü—ñ–¥–≤–∏–ø–∞–¥–æ–∫ 2–∞: –ø–æ–≤–Ω—ñ—Å—Ç—é –∫–æ–ø—ñ—é—î–º–æ
                Z[k] = Zkp
            else:
                # –ü—ñ–¥–≤–∏–ø–∞–¥–æ–∫ 2–±: –ø–æ—Ä—ñ–≤–Ω—é—î–º–æ –¥–∞–ª—ñ –∑–∞ –º–µ–∂–∞–º–∏ r
                nmatch = 0
                for i in range(r + 1, len(s)):
                    if s[i] == s[i - k]:
                        nmatch += 1
                    else:
                        break
                l, r = k, r + nmatch
                Z[k] = r - k + 1

    return Z

 # –Ü–Ω—ñ—Ü—ñ–∞–ª—ñ–∑—É—î–º–æ —Å–ø–∏—Å–æ–∫ occurrences –¥–ª—è –∑–±–µ—Ä—ñ–≥–∞–Ω–Ω—è –ø–æ–∑–∏—Ü—ñ–π, –¥–µ –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è –∑–±—ñ–≥ –∑ —à–∞–±–ª–æ–Ω–æ–º.

        if Z[i] == len(p):

 # –ü–æ–∑–∏—Ü—ñ—è —Ä–æ–∑—Ä–∞—Ö–æ–≤—É—î—Ç—å—Å—è —è–∫ i ‚Äì (len(p) + 1), –æ—Å–∫—ñ–ª—å–∫–∏ –º–∏ —Ä–æ–∑–≥–ª—è–¥–∞—î–º–æ —Ä–æ–∑—à–∏—Ä–µ–Ω–∏–π —Ä—è–¥–æ–∫ s,

    return occurrences

    for i in range(1, len(s)): 
        # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —á–∏ –≤–∏–∫–æ–Ω—É—é—Ç—å—Å—è —É–º–æ–≤–∏ –¥–ª—è –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ–≥–æ —Å—Ç–∏—Å–Ω–µ–Ω–Ω—è 
        if (i + z_vec[i] == len(s)) and (z_vec[i] % i == 0):

            return s[:i]
